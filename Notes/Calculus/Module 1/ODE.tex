\subtopic{ODE}
A differential equation involving derivortives of dependent variable with respect to only one independent variable is called ODE

e.g:
$$
	3\left(\frac{d^{2} y}{dx^2 }\right)^{4}+\left(\frac{dy}{dx}\right)^{5}=0
$$

This has degree 4 and order 2

\ssubsection{\texorpdfstring{1$^{\text{st}}$ order differential equation}{1st order differential equation}}
\begin{flalign*}
	\text{General form} & : f\left(x, y, y'\right)=0        & \\
	                    & \text{ or } y'=g(x, y)            & \\
	\\
	                    & \text{ e.g: } y^2=\sqrt{x^2 +y^2}
\end{flalign*}


\ssubsection{\texorpdfstring{1$^{\text{st}}$ Order Linear Differential equation}{1st Order Linear Differential equation}}
General form:
\begin{flalign*}
	 & y'+P(x) y=Q(x) \hspace*{1cm} \text{where functions } P \& Q \text{ are continuous in some interval } \mathtt{I} \\
\end{flalign*}

General solution:
\begin{flalign*}
	y F            & =\int F Q dx + C \\
	\\
	\text{where F} & = e^{\int P dx}
\end{flalign*}

F is also known as \term{Integration factor}

\multiskip{3}

\qa{1}{solve $y'+2 x y=x$}{

	\twosided{.52}{.35}{
		Given equation is an $1^{\text{st}}$ order LDE
		comparing the general form: $y'+P(x) y=Q(x)$ we get:
		$P = 2x, Q=x$
		\begin{flalign*}
			\therefore \text{Integrating factor } F & =e^{\int P \cdot dx} = e^{\int 2x \cdot dx} = e^{x^2 } & \\
			\therefore \text{Solution: } y\cdot F   & =\int F Q dx + C                                         \\
			y e^{x^2 }                              & =\int e^{x^2 } x\ dx + C
		\end{flalign*}

		let $t = x^2$
		\\

		$\therefore 2x = \dfrac{dt}{dx}$\ \ or\ \ $x dx = \dfrac{1}{2}dt$
	}{\begin{flalign*}
			\therefore \int e^{x^2 } x\ dx & = \int \frac{1}{2} e^t dt               & \\
			                               & = \frac{1}{2} e^t = \frac{1}{2} e^{x^2} &
		\end{flalign*}
		\begin{flalign*}
			\therefore y e^{x^2 } & = \frac{1}{2} e^{x^2 } + C                                  & \\
			\therefore y          & = e^{-x^2 }\left(\frac{1}{2} e^{x^2 } + C\right)            & \\
			\\
			                      & \boxed{y                       = \frac{1}{2} + e^{-x^2 } C} &
		\end{flalign*}
	}
}

\qa{2}{Solve\ \ $\dfrac{dy}{dx} + 2y\tan(x)=\sin(x)$}{

	\twosided{.5}{.4}{
		Here $P(x) = 2 \tan(x), Q(x)=\sin(x)$

		\begin{flalign*}
			\text{Integrating factor (IF) } F & =e^{\int P \cdot dx}                     & \\
			                                  & =e^{\int 2 \tan(x) dx}                     \\
			                                  & =e^{2 \cdot \ln |\sec (x)|}                \\
			                                  & =\left(e^{\cdot \ln |\sec (x)|}\right)^2   \\
			                                  & =|\sec (x)|^{2} \equiv \sec ^{2}(x)
		\end{flalign*}
	}{
		$\therefore$ Solution:
		\begin{flalign*}
			y \cdot \sec^{2}(x) & =\int \sin(x) \cdot \sec ^{2}(x) dx+C               & \\
			                    & =\int \cdot \sec (x) \tan(x) dx+C                     \\
			                    & =\sec (x)+C                                           \\
			\therefore y        & =\frac{\sec(x)}{\sec^{2}(x)}+ \frac{C}{\sec^{2}(x)}   \\
			y                   & =\cos(x)+\cos^2(x) \cdot C
		\end{flalign*}
	}

}

\qa{3}{
	Find solution of initial value problem:
	\eqnleft{\hspace*{2cm}x^2 y'-x y=x^4 \cos(2x), \hspace*{2cm}y(\pi)=2 \pi}
}{

	\twosided{.45}{.45}{
		\begin{flalign*}
			\quad x^2 y'-xy            & =x^4 \cos(2 x) & \\
			\therefore y_{-}'-x^{-1} y & =x^2 \cos(2 x) &
		\end{flalign*}
		\begin{flalign*}
			P(x)                  & =-x^{-1}, Q(x)=x^2 \cos(2 x)                 & \\
			\therefore \text{I.F} & =  e^{\int p(x) \cdot dx}=e^{-\int x^{-1}dx} & \\
			                      & =e^{-\ln(x)}=x^{-1}=\frac{1}{x}
		\end{flalign*}

		General Solution:
		\begin{flalign*}
			y \cdot I F         & =\int Q \cdot I F \cdot dx+C                            & \\
			y \cdot \frac{1}{x} & =\int x^2 \frac{1}{x} \cos(2 x) \cdot dx  + C             \\
			                    & =\int x \cos(2 x) dx + C                                  \\
			                    & =\frac{x \sin(2x)}{2}-\frac{1}{2} \int \sin(2 x) dx + C   \\
			                    & =\frac{x \sin(2 x)}{2}+\frac{\cos(2 x)}{4} + C
		\end{flalign*}
		\eqnleft{\therefore y =\frac{x^2 \cdot \sin(2 x)}{2}+\frac{x \cos(2 x)}{4}+x C \tag*{(1)} \label{eq:3y} &}
	}{
		It is given that $y(\pi)=2 \pi$

		\begin{flalign*}
			\therefore 2 \pi & = \frac{\pi^{2} \cdot \overbrace{\sin(2 \pi)}^{0}}{2} + \frac{\pi \cdot \cos(2 \pi)}{4}+\pi \cdot C & \\
			\therefore 2     & = \frac{1}{4}+c                                                                                     & \\
			                 & \Rightarrow c=2 - \frac{1}{4}=\frac{7}{4}                                                           & \\
		\end{flalign*}

		Substitute in Eq.\ref{eq:3y} to get specific solution: \\

		\fbox{$y=\dfrac{x^2 \sin(2 x)}{2}+\dfrac{x\cos(2 x)}{4}+\dfrac{7}{4}x$}
	}
}

\qa{4}{Solve $y'-2xy=2$}{

\twosided{.4}{.5}{
Let $P(x)=-2x, Q(x)=2x$
\eqnleft{I.F =e^{\int-2 x \cdot dx}=e^{-x^2 }}
$\therefore$ Solution: $y \cdot e^{-x^2 }=\int 2 x \cdot e^{-x^2 } dx+c$
}{
Let $-x^2 =t \Rightarrow dx= \dfrac{dt}{-2x}$
\begin{flalign*}
	\therefore  \int 2x e^{-x^2}dx + C = -\int e^t dt & = -e^t + C       & \\
	                                                  & = -e ^{-x^2} + C   \\
\end{flalign*}
\begin{flalign*}
	\therefore y\cdot e^{-x^2 } & =-e^{-x^2 } \cdot C    & \\
	\Aboxed{\therefore y        & =-1 + e^{x^2} \cdot C}
\end{flalign*}
}
}

\qa{5}{Solve $xy'-2 y=-x$}{

	\twosided{.3}{.4}{
		\begin{flalign*}
			xy'-2 y                          & =-x & \\
			\therefore y'-\frac{2}{x}\cdot y & =-1
		\end{flalign*}
		\begin{flalign*}
			\therefore P(x) & =-\frac{2}{x}, Q(x)=-1                   & \\
			IF              & = e^{\int P x}=e^{\int -2 / x \cdot d x} & \\
			                & =e^{-2 \ln (x)}=x^{-2}
		\end{flalign*}

	}{
		$\therefore$ Solution:
		\begin{flalign*}
			y \cdot x^{-2} & =\int x^{-2} \times (-1) \cdot dx + C & \\
			               & =\frac{x^{-1}}{-1} \times -1 + C        \\
			               & =\frac{1}{x} + C
		\end{flalign*}

		$\therefore$ \fbox{$y=x + C x^2$}

	}
}

\qa{6}{Solve $x y'+2 y=\frac{\cos(x)}{x}$}{

	\twosided{.4}{.5}{
		\begin{flalign*}
			x y'+2 y               & =\frac{\cos(x)}{x}    & \\
			y'+\frac{2}{x} \cdot y & =\frac{\cos(x)}{x^2 }
		\end{flalign*}
		\begin{flalign*}
			P(x)       & =\frac{2}{x}, Q(x)=\frac{\cos(x)}{x^2 } & \\
			\text{IF } & =e^{\int P \cdot dx} = e^{\int 2/x\ dx} & \\
			           & =e^{\ln \left(x^2 \right)}=x^2
		\end{flalign*}

	}{
		Solution: $y \cdot I F=\int I F \cdot Q \cdot dx+C$
		\begin{flalign*}
			y \cdot x^2 & =\int x^2 \cdot \frac{\cos(x)}{x^2 } \cdot dx+C & \\
			            & =\int \cos(x) dx+C                                \\
			y \cdot x^2 & =\sin(x)+C                                        \\
			\Aboxed{y   & =\frac{\sin(x)}{x^2 }+\frac{C}{x^2 }}
		\end{flalign*}
	}
}

\qa{7}{Solve $y'+\dfrac{2 y}{x}=\dfrac{4}{x}$. where $y(1) = 6$}{

	\twosided{.35}{.4}{
		\begin{flalign*}
			P(x)       & =\dfrac{2}{x}, Q(x)=\dfrac{4}{x}           & \\
			\text{IF } & =e^{\int P dx}=e^{\int 2 / x \cdot dx}=x^2
		\end{flalign*}
		$\therefore$ Solution:
		\begin{flalign*}
			y \cdot x^2  & =\int x^2 \cdot \frac{4}{x} dx+C=2 x^2 +C & \\
			\therefore y & =2+\frac{C}{x^2}
		\end{flalign*}
	}
	{
		Given $y(1)=6$
		\eqnleft{\therefore \quad 6=2+\frac{C}{1} \Rightarrow C=4}
		\eqnleft{y=2+\frac{4}{x^2}}
	}
}
