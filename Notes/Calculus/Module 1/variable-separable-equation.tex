\topic{Variable Separable Equation}
A differential equation of the form $m(x, y) dx+n(x, y) dy=0$ is a variable separable equation if it can be expressed in the form: $f(x) dx+g(y) d y=0$

\multiskip{2}

\qa{1}{
	Solve : $\frac{dy}{dx}=\frac{y}{x}$
}{

	\begin{flalign*}
		 & \frac{dy}{dx}=\frac{y}{x}             & \\
		 & \Rightarrow dy \cdot x=dx \cdot y       \\
		 & \Rightarrow \frac{dx}{x}=\frac{dy}{y}
	\end{flalign*}

	Integrate both sides:
	\begin{flalign*}
		\int \frac{dx}{x} & = \int \frac{dy}{y}       & \\
		\ln(y)            & = \ln(x) + \ln(C)           \\
		                  & = \ln(xC)                   \\
		                  & \Rightarrow \fbox{y = xC}
	\end{flalign*}
}


\qa{2}{
	Solve: $(y + 2)dx + y(x + 4)dy = 0$
}{
	Divide by $(y+2)(x+4)$

	\eqnleft{\frac{1}{x+4}dx + \frac{y}{y+2}dy = 0}
	Integrating both sides:

	\begin{flalign*}
		\int \frac{dx}{x+4} & + \int \frac{y}{y+2}dy                                 & = \ln(C) & \hspace*{1cm} \text{[ln(C) is used to make further steps easier]} \\
		\int \frac{dx}{x+4} & + \int \frac{y + 2 -2}{y+2}dy                          & = \ln(C) &                                                                   \\
		\int \frac{dx}{x+4} & + \int \left[\frac{y+2}{y+2} + \frac{-2}{y+2}\right]dy & = \ln(C) &                                                                   \\
		\int \frac{dx}{x+4} & + \int \left[1 + \frac{-2}{y+2}\right]dy               & = \ln(C) &
	\end{flalign*}
	\begin{flalign*}
		\ln(x+4) & + y - 2\ln(y + 2) = \ln(C)                                      \\
		         & y = \ln(C) +  2\ln(y + 2) - \ln(x+4)                            \\ \\
		         & \therefore y = \ln\left[C \cdot \frac{(y+2)^2}{x+4}\right] &  &
	\end{flalign*}
}


\qa{3}{
	solve $3 x \sin(y) \cdot dx+\left(x^2 +1\right) \cdot \cos(y) \cdot dy=0$
}{

	Divide by $\sin(y) \cdot\left(x^2 +1\right)$

	\begin{flalign*}
		 & \therefore \frac{x}{x^2 +1} dx+\frac{\cos(y)}{\sin(y)} dy=0       & \\
		 & \text{Integrate both sides:}                                      & \\ \\
		 & \int \frac{x}{x^2 +1}dx + \int \frac{\cos(y)}{\sin(y)} dy=\ln (C)
	\end{flalign*}
	\begin{flalign*}
		\text{Let } t & =x^2+1   \Rightarrow dx =\dfrac{dt}{2x} & \\
		u             & =\sin(y)  \Rightarrow du = dy\cos(y)    &
	\end{flalign*}
	\begin{flalign*}
		\therefore \int \frac{dt}{2t}        + \int \frac{du}{u} & =\ln (C) & \\
		\frac{1}{2} \ln (t)                  +\ln (u)            & =\ln (C)   \\
		\frac{1}{2} \ln \left(x^2 +1\right)  +\ln (\sin(y))      & =\ln (C)
	\end{flalign*}
	\begin{flalign*}
		\ln(\sin(y)) & = \ln\left[\frac{C}{\left(x^2+1\right)^2}\right]       & \\
		\sin(y)      & = \frac{C}{\left(x^2+1\right)^2}                         \\
		\text{or}                                                               \\
		y            & = \sin^{-1}\left[\frac{C}{\left(x^2+1\right)^2}\right]   \\
	\end{flalign*}
}


\qa{4}{Solve $\tan(\theta) dr+2 r \cdot d\theta=0$}
{
	\begin{flalign*}
		\frac{dr}{2 r}+\frac{d\theta}{\tan(\theta)}                                               & = 0                                      & \\
		\int \frac{dr}{2 r}+\int \underbrace{\frac{d\theta}{\tan(\theta)}}_{\cot(\theta) d\theta} & =\ln (C)                                   \\
		\frac{1}{2} \ln (r)+\ln (\sin(\theta))                                                    & =\ln (C)                                   \\
		\ln (\sqrt{x})+\ln (\sin(\theta))                                                         & =\ln (C)                                   \\
		\ln (\sqrt{r})                                                                            & =\ln \left[\frac{C}{\sin(\theta)}\right]   \\
		\sqrt{r}                                                                                  & =\frac{C}{\sin(\theta)}                    \\
		\Rightarrow r                                                                             & = \frac{C}{\sin^{2}(\theta)}               \\
		r                                                                                         & = C\cdot \csc^2(\theta)
	\end{flalign*}
}

\qa{5}{Solve $4 x y\ dx+\left(x^2 +1\right) dy=0$}{

	\begin{flalign*}
		4 x y\ dx+\left(x^2 +1\right) dy               & =0                                      & \\
		\Rightarrow \frac{4 x}{x^2 +1} dx+\frac{dy}{y} & =0                                        \\
		2\int \frac{2x}{x^2 + 1}dx + \int \frac{dy}{y} & = \ln(C)                                  \\
		2\ln(x^2 + 1) + \ln(y)                         & = \ln(C)                                  \\
		                                               & \Rightarrow y = \frac{C}{(x^2 + 1) ^ 2}
	\end{flalign*}
}
