\topic{Homogeneous Differential Equation}
An differential equation that can be reduced into the form: $\dfrac{dy}{dx}=F\left(\dfrac{y}{x}\right)$ is called homogerious differential equation.
This can be solved by putting $y=v x$ and hence reducing to variable separable form.

\bigskip

\qa{1}{
	Solve $2 x y \cdot \frac{dy}{dx}-y^2+x^2 =0$
}{
	\twosided{.5}{.3}{
		\begin{flalign*}
			2xy \frac{dy}{dx}-y^2+x^2 & = 0                  & \\
			\Rightarrow \frac{dy}{dx} & =\frac{y^2-x^2}{2xy}
		\end{flalign*}
		Put $y=vx$ $\therefore v=\frac{y}{x}$

		$\therefore$ Equation becomes:
		\begin{flalign*}
			v+x \cdot \frac{dv}{dx}                             & =\frac{v^{2} x^2 -x^2 }{2 x^2 v}                   & \\

			\therefore x \cdot \frac{dv}{dx}                    & =\frac{v^{2}-1}{2 v}-v=\frac{v^{2}-1-2 v^{2}}{2 v}   \\
			x \cdot \frac{dv}{dx}                               & =\frac{-\left(1+v^{2}\right)}{y v}                   \\
			\frac{2 v}{\left(c 1+v^{2}\right)} \cdot dv         & =\frac{dx}{x}                                        \\
			\therefore-\int \frac{2 v}{\left(1+v^{2}\right)} dv & =\int \frac{dx}{x}+\ln (c)                           \\
			                                                    & =-\ln \left(v^{2}+1\right)=\ln (x)+\ln (c)           \\
			x \equiv \ln \left(v^{2}+1\right)                   & =-\ln (x)+\ln (c)                                    \\
			\therefore v^{2}+1                                  & =\frac{c}{x}                                         \\
			\frac{y^2}{x^2 }+1                                  & =\frac{c}{x} \Rightarrow y^2+x^2 =c x                \\
		\end{flalign*}
