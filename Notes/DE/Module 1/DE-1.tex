\ussubsection{Decimal to binary}

eg: convert $(12.05)_{10}$ to binary

\begin{minipage}[]{.5\textwidth}
	\begin{tblr}{
			colspec={l|l},
		}
		2                  & 12 &  &   \\ \cline{2-2}
		2                  & 6  &  & 0 \\ \cline{2-2}
		2                  & 3  &  & 0 \\ \cline{2-2}
		\SetCell[c=2]{r} 1 &    &  & 1 \\
	\end{tblr}

	\bigskip

	$\therefore (12)_{10} = (1100)_2$
\end{minipage}
\begin{minipage}[]{.4\textwidth}
	\begin{tblr}{rl}
		0.15 $\times$ 2 & = & 0 & .3 \\
		0.3  $\times$ 2 & = & 0 & .6 \\
		0.6  $\times$ 2 & = & 1 & .2 \\
		0.2  $\times$ 2 & = & 0 & .4 \\
	\end{tblr}

	\bigskip

	$\therefore (.05)_{10} \approx (.0010)_2$
\end{minipage}

\bigskip

$\therefore (12.05)_{10} \approx (1100.0010)_2$
