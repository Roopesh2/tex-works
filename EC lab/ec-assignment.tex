\usection{Zener Series Regulator}
\vspace*{1cm}
\linespread{.9}
\usubsection{Aim}
\begin{enumerate}
	\item To design a 5V Zener series regulator for a input voltage of +8 to +20V and maximum load current of 50mA
	\item Plot and its load regulation and line regulation and find percentage load regulation and line regulation.
\end{enumerate}

\usubsection{Design}
\begin{itemize}
 \item $V_{in}(max) = 20V$
 \item $V_Z = 5.1V$ \hspace*{.5cm} $V_{in}(min) = 8V$
 \item $I_L(max) = 50mA$ \hspace*{.5cm} $I_L(min) = 10mA$
 \item $I_Z(\max) = 20mA$ \hspace*{.5cm} $I_Z(min) = 10mA$
\end{itemize}

$V_s(max) = V_{in}(max) - V_Z = 20V - 5.1V = \res{14.9V}$

\vspace*{.5cm}
At minimum load current:

$$R_s{max} = \dfrac{V_s(max)}{I_L(min) + I_Z(max)} = \dfrac{14.9V}{10mA + 20mA} \approx \res{500 $\Omega$} $$

\vspace*{.5cm}
At maximum load current:

$$R_s{min} = \dfrac{V_s(max)}{I_L(max) + I_Z(min)} = \dfrac{14.9V}{50mA + 10mA} \approx \res{250 $\Omega$} $$

\vspace*{.5cm}
$\therefore R_s = \sqrt{R_s{max}\times R_s{min}} = \sqrt{500\times 250} = \res{353 $\Omega$}$
